TTITLE OFF

ALTER SESSION SET NLS_DATE_FORMAT = 'DD-MON-YY';
SET LINESIZE 200
SET PAGESIZE 200
COLUMN Booking_Id FORMAT A10;
COLUMN Booking_Days FORMAT 999;
COLUMN Booking_Start_Date FORMAT A18;
COLUMN Booking_End_Date FORMAT A16;
COLUMN Route_Id FORMAT A8;
COLUMN Start_Location FORMAT A30;
COLUMN End_Location FORMAT A30;
COLUMN Distance FORMAT 999.999;
CL SCR

ACCEPT MONTH CHAR PROMPT 'Enter month (e.g., MON): ';

TTITLE ' Route and Booking Information in ' '&MONTH' '-24 Report ' SKIP 3;

SELECT B.Booking_Id, B.Booking_Days, B.Booking_Start_Date, B.Booking_End_Date, R.Route_Id, R.Start_Location, R.End_Location, R.Distance
FROM BOOKING B JOIN ROUTE R
ON R.Route_Id = B.Route_Id
WHERE TO_CHAR(B.Booking_Start_Date, 'MON-YY') = UPPER('&MONTH-24')
ORDER BY R.Route_Id, B.Booking_Days desc;
