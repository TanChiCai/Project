Create table bus(
   	Bus_Id  varchar(10)  not null,
   	License_Plate  varchar(10)  not null,
   	Road_Tax_Expiry_date  date  not null,
   	Maintenance_Status  varchar(20) DEFAULT 'ACTIVE' , 
	Bus_Type_Id  varchar(10)  not null,
Primary key(Bus_Id),
Foreign key(Bus_Type_Id) references BUS_TYPE(Bus_Type_Id),
Constraint chk_maintenance_status check (
upper(Maintenance_Status) in ('OPERATIONAL', 'UNDER MAINTENANCE'))
);
