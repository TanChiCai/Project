Create table SCHOOL(
   	School_Id  varchar(10)  not  null,
   	School_Name  varchar(30)  not null,
   	School_Address  varchar(50)  not null,
   	School_Email  varchar(30) not null,
   	School_Contact  varchar(14) not null,
Primary key(School_Id),
Constraint chk_School_Email check(
REGEXP_LIKE(School_Email,'^[a-zA-Z]\w+@(\S+)$')
),
Constraint chk_School_Contact check (
REGEXP_LIKE(School_Contact, '^60[1-9]{1}-[0-9]{7}$')
)
);
