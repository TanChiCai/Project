Create table ROUTE(
   	Route_Id  varchar(10)  not null,
   	Start_Location  varchar(45)  not null,
   	End_Location  varchar(45)  not null,
   	Distance  number(5,2)  not null,
	Distance_Price  number(7,2) not null,
Primary key(Route_Id),	
Constraint chk_distance check (Distance > 0),
Constraint chk_distance_price check (Distance_Price > 0) 
);
